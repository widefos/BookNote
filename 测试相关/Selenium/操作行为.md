我们可以把这些操作分为四大类：

1. **浏览器操作** - 控制浏览器本身

2. **元素操作** - 与页面元素交互

3. **表单操作** - 填写和提交表单

4. **高级操作** - 鼠标和键盘动作

### 2. 元素操作

#### **click() - 点击元素**

```
# 点击按钮、链接、复选框等
login_button = driver.find_element(By.ID, "login-btn")
login_button.click()

# 点击复选框
checkbox = driver.find_element(By.NAME, "agree-terms")
checkbox.click()
```


#### **send_keys() - 输入文本**

```
# 在输入框中输入文本
username_input = driver.find_element(By.ID, "username")
username_input.send_keys("my_username")

# 先清空再输入（重要！）
search_box = driver.find_element(By.NAME, "q")
search_box.clear()           # 清空原有内容
search_box.send_keys("新的搜索关键词")
```

#### **clear() - 清空输入框**

```
email_input = driver.find_element(By.ID, "email")
email_input.clear()  # 清空输入框
```

#### **get_attribute() - 获取元素属性**

```
# 获取元素的各类属性
element = driver.find_element(By.ID, "submit-btn")

# 获取标准属性
class_name = element.get_attribute("class")
id_value = element.get_attribute("id")
value = element.get_attribute("value")
href = element.get_attribute("href")  # 对于链接

# 获取自定义属性
data_id = element.get_attribute("data-id")
```


#### **text - 获取元素可见文本**
```
# 获取元素显示的文本内容
header = driver.find_element(By.TAG_NAME, "h1")
print(header.text)  # 输出：欢迎来到我的网站

success_message = driver.find_element(By.CLASS_NAME, "alert-success")
if "成功" in success_message.text:
    print("操作成功了！")
```
### 3. 表单操作

#### **提交表单**

```
# 方法1：找到提交按钮并点击
submit_button = driver.find_element(By.XPATH, "//button[@type='submit']")
submit_button.click()

# 方法2：在任意表单元素上调用submit()
form_element = driver.find_element(By.TAG_NAME, "form")
form_element.submit()

# 方法3：在输入框按回车
password_input = driver.find_element(By.ID, "password")
password_input.send_keys("mypassword")
password_input.submit()  # 相当于按回车
```

#### **选择下拉列表**

```
from selenium.webdriver.support.ui import Select

# 定位下拉列表
dropdown = Select(driver.find_element(By.ID, "country"))

# 选择方式1：按可见文本
dropdown.select_by_visible_text("中国")

# 选择方式2：按value属性
dropdown.select_by_value("cn")

# 选择方式3：按索引（从0开始）
dropdown.select_by_index(1)

# 获取当前选中的选项
selected_option = dropdown.first_selected_option
print(f"当前选择: {selected_option.text}")

# 获取所有选项
all_options = dropdown.options
for option in all_options:
    print(option.text)
```



### 4. 高级操作


#### **鼠标悬停**

```
from selenium.webdriver.common.action_chains import ActionChains

menu_element = driver.find_element(By.ID, "main-menu")
submenu_element = driver.find_element(By.ID, "sub-menu")

# 创建动作链
actions = ActionChains(driver)

# 鼠标悬停到主菜单，然后点击子菜单
actions.move_to_element(menu_element).click(submenu_element).perform()

# 其他鼠标操作
# actions.double_click(element).perform()  # 双击
# actions.context_click(element).perform() # 右键点击
# actions.drag_and_drop(source, target).perform() # 拖放
```


#### **键盘操作**

```
from selenium.webdriver.common.keys import Keys

search_box = driver.find_element(By.NAME, "q")

# 输入文本并按键
search_box.send_keys("Selenium")
search_box.send_keys(Keys.ENTER)  # 按回车键

# 组合按键
search_box.send_keys(Keys.CONTROL + "a")  # Ctrl+A 全选
search_box.send_keys(Keys.BACKSPACE)      # 删除

# 常用特殊键：
# Keys.TAB, Keys.ESCAPE, Keys.ARROW_UP, Keys.ARROW_DOWN, etc.
```

#### **执行JavaScript**


```
# 滚动到页面底部
driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")

# 滚动到特定元素
element = driver.find_element(By.ID, "footer")
driver.execute_script("arguments[0].scrollIntoView();", element)

# 修改元素属性（比如让隐藏元素显示）
driver.execute_script("document.getElementById('hidden-element').style.display = 'block';")

# 点击被遮挡的元素
button = driver.find_element(By.ID, "obscured-button")
driver.execute_script("arguments[0].click();", button)
```



### 完整实战示例：自动化登录流程

```
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.common.keys import Keys
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

driver = webdriver.Chrome()

try:
    # 1. 打开登录页面
    driver.get("https://example.com/login")
    
    # 2. 等待页面加载完成
    wait = WebDriverWait(driver, 10)
    
    # 3. 输入用户名
    username_input = wait.until(EC.presence_of_element_located((By.ID, "username")))
    username_input.clear()
    username_input.send_keys("testuser")
    
    # 4. 输入密码
    password_input = driver.find_element(By.ID, "password")
    password_input.send_keys("testpass123")
    
    # 5. 点击"记住我"复选框（如果未选中）
    remember_me = driver.find_element(By.ID, "remember-me")
    if not remember_me.is_selected():
        remember_me.click()
    
    # 6. 提交表单（按回车键）
    password_input.send_keys(Keys.ENTER)
    
    # 7. 等待登录成功并验证
    welcome_message = wait.until(
        EC.visibility_of_element_located((By.ID, "welcome-msg"))
    )
    
    # 断言登录成功
    assert "欢迎" in welcome_message.text
    assert "dashboard" in driver.current_url
    
    print("登录测试通过！")
    
finally:
    driver.quit()
```
