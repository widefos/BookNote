


```
-- 用户表
CREATE TABLE users (
    id BIGINT PRIMARY KEY,
    username VARCHAR(50),
    password VARCHAR(100)
);

-- 角色表（注意角色名通常存 'ROLE_ADMIN' 或 'ADMIN'）
CREATE TABLE roles (
    id BIGINT PRIMARY KEY,
    name VARCHAR(50)  -- 内容如：'ADMIN', 'USER', 'MANAGER'
);

-- 用户-角色关联表
CREATE TABLE user_roles (
    user_id BIGINT,
    role_id BIGINT,
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (role_id) REFERENCES roles(id)
);
```
